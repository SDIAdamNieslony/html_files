fcm_gen_load function is a load generator for other FCM Live Scripts

Don't change it.
function z = fcm_gen_load(loadNr)
rng('default')
switch loadNr
case 1 % scaled signal based on 20 turning poins and with 0 mean
fs = 512;
X = [-2 1 -3 5 -1 3 -4 4 -2 1 -3 5 4 6];
lX = length(X)-1;
Y = -diff(X)/2.*cos(pi*(0:1/fs:1-1/fs)') + (X(1:lX)+X(2:lX+1))/2;
Y = [Y(:);X(end)];
z = Y;
%z = [(0:1/fs:lX)' Y];
%plot(0:lX,X,'o',0:1/fs:lX,Y)
case 2 % scaled 10000 random numbers with 0 mean
z = 100*randn(10000,1);
case 3 % scaled 10000 random numbers with 0 mean, non-stationary signal
x = 150*randn(10000,1);
r = 0.3 + 1.5*rand(10,1);
z = x.*interp1(linspace(0,1,length(r)),r,linspace(0,1,length(x)))';
case 4
a = randn(20,3);
t = cumsum(1+rand(20,3));
tz = linspace(max(t(1,:)),min(t(end,:)),500);
z = zeros(length(tz),3);
for i=1:3
z(:,i) = interp1(t(:,i), a(:,i), tz, 'pchip');
end
otherwise
error('Load case number unknow!')
end
end
